rm -rf .next && npm run build && git add . && git commit -m "production build" && git push

# Full clean production build
rm -rf .next && rm -rf node_modules/.cache && npm run build && npm run start

git add . && git commit -m "testing" && git push

git add src/ && git commit -m "updated components" && git push

git status && git add . && git commit -m "changes" && git push
